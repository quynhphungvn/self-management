@startuml
entity video {
*id: int 
*title: varchar(100) NOT NULL
*subtitle: text
*url: varchar(100) NOT NULL
*group_id: int NOT NULL
*user_id: int NOT NULL
*view_count: int
*UNIQUE(url, group_id, user_id) 
}
entity video_group {
*id: int 
*name: varchar(30) NOT NULL
*user_id: int NOT NULL
*UNIQUE(user_id, name)
}
entity user {
*id: int
*username: varchar(100) NOT NULL UNIQUE 
*password: varchar(32)
}
entity symbol_view_info {
*id: int
*user_id: int
*symbol_id: int
*view_count: int
}
entity word {
*id: int 
*word: varchar(25) NOT NULL UNIQUE
*phonetic: varchar(20)
*user_add: boolean
}
entity word_review {
*id: int 
*word_id: int 
*user_id: int 
*known: boolean
}
entity symbol {
*id: int 
*symbol: varchar(10) NOT NULL UNIQUE
*example: varchar(20)
*example_phonetic: varchar(20)
*video_guide_url: varchar(100)
*image_guide_url: varchar(100)
*type: varchar(15)
}
user ||--o{ video_group
user ||--o{ word_review
user ||--|| symbol_view_info
user ||--o{ video
symbol ||--|| symbol_view_info
word ||--o{ word_review
video_group ||--o{ video

@enduml
